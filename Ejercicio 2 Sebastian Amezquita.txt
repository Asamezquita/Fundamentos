package clase;
import java.util.Scanner;
public class matriz {

	public static void main(String[]args){
		Scanner leer = new Scanner(System.in);
		int n;
		
		System.out.print("Digite un numero:");
		n= leer.nextInt();
		for (int i=0;i<n;i++) {
		for (int j=0;j<n;j++) {
			System.out.print("*");
		}
		System.out.println("*");
		}
	}
}
